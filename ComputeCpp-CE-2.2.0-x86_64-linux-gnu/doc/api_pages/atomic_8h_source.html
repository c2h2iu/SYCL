287 template <typename T, access::address_space addressSpace>
288 inline T atomic<T, addressSpace>::fetch_min(const T operand,
289  const memory_order mem_order) {
290  /* Standard C++ 11 defines no "min" operation, so this function emulates the
291  * behaviour by executing a loop. First, the value is loaded, then compared
292  * against the operand. After this a compare_exchange is used to check that
293  * the value hasn't been updated sneakily (by another thread say); if it has
294  * (i.e. m_data doesn't equal its old value) store the new value of the atomic
295  * and try again. */
296  T old = m_data->load(std::memory_order_relaxed);
297  do {
298  if (old < operand) {
299  break;
300  }
301  } while (!m_data->compare_exchange_weak(
302  old, operand, static_cast<std::memory_order>(mem_order)));
303  return old;
304 }
305 
306 template <typename T, access::address_space addressSpace>
307 inline T atomic<T, addressSpace>::fetch_max(const T operand,
308  const memory_order mem_order) {
309  /* Standard C++ 11 defines no "max" operation, so this function emulates the
310  * behaviour by executing a loop. First, the value is loaded, then compared
311  * against the operand. After this a compare_exchange is used to check that
312  * the value hasn't been updated sneakily (by another thread say); if it has
313  * (i.e. m_data doesn't equal its old value) store the new value of the atomic
314  * and try again. */
315  T old = m_data->load(std::memory_order_relaxed);
316  do {
317  if (operand < old) {
318  break;
319  }
320  } while (!m_data->compare_exchange_weak(
321  old, operand, static_cast<std::memory_order>(mem_order)));
322  return old;
323 }
