497  accessor<elemT, 1, kMode, kTarget, access::placeholder::true_t>
498  get_accessor_with_added_offset(int addedOffset) const {
499  detail::access_range range = this->get_access_range();
500  // ensure that negative values move the offset backwards
501  intptr_t result = static_cast<intptr_t>(addedOffset) +
502  static_cast<intptr_t>(range.offset[0]);
503  // check underflow
504  if (result < 0) {
505  COMPUTECPP_CL_ERROR_CODE_MSG(
506  CL_SUCCESS, detail::cpp_error_code::ACCESSOR_ARGUMENTS_ERROR, nullptr,
507  "Attempted arithmetic operation out of accessor bounds. Underflow");
508  }
509  // check overflow
510  size_t resultSizeT = static_cast<size_t>(result);
511  if (resultSizeT + range.range[0] > this->get_store_range()[0]) {
512  COMPUTECPP_CL_ERROR_CODE_MSG(
513  CL_SUCCESS, detail::cpp_error_code::ACCESSOR_ARGUMENTS_ERROR, nullptr,
514  "Attempted arithmetic operation out of accessor bounds. Overflow");
515  }
516  // set the result
517  range.offset[0] = resultSizeT;
518 
519  cl::sycl::storage_mem storage(this->get_store());
520  return accessor<elemT, 1, kMode, kTarget, access::placeholder::true_t>(
521  storage, range);
522  }
