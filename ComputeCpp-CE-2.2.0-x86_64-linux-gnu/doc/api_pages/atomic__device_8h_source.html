295 // OpenCL 1.2 has no store operation, so swap with the desired value, then
296 // discard the old value (i.e. *m_data).
297 template <typename T, access::address_space AddressSpace>
298 inline void atomic<T, AddressSpace>::store(T operand, memory_order mem_order) {
299  detail::atomic_helper<T>::xchg(m_data, operand);
300 }

313 // OpenCL 1.2 does not implement compare_and_exchange in the same way as C++11,
314 // so this function emulates the behavior on the device. It attempts to figure
315 // out if the call to cmpxchg actually altered the atomic value by capturing the
316 // old value then comparing it to expected. If they are the same, then the
317 // comparison will have succeeded, so we can return true. If they aren't, then
318 // comparison failed, so we need to update expected manually then return false.
319 // The helper function performs this part of the logic, since it is shared by
320 // all compare_exchange device implementations.
321 template <typename T, access::address_space AddressSpace>
322 inline cl_bool atomic<T, AddressSpace>::compare_exchange_strong(
323  T& expected, T desired, memory_order success, memory_order fail) {
324  auto old = detail::atomic_helper<T>::cmpxchg(m_data, expected, desired);
325  return ::cl::sycl::detail::cmpxchg_helper(old, expected);
326 }
