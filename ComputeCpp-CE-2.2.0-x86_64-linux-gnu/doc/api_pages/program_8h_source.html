507  template <typename kernelT>
508  static program create_program_for_kernel(cl::sycl::context c) {
509  if (detail::kernel_info<kernelT>::name == nullptr && !c.is_host()) {
510  COMPUTECPP_CL_ERROR_CODE_MSG(
511  CL_SUCCESS, detail::cpp_error_code::KERNEL_NOT_FOUND_ERROR,
512  c.get_impl().get(),
513  "Unable to retrieve kernel function, is integration header included?")
514  } else if (c.is_host()) {
515  return program(c);
516  }
517  // Otherwise, create a normal Program
518  std::string kernelName = detail::kernel_info<kernelT>::name;
519  const detail::kernel_binary_info& binInfo =
520  *program::select_kernel_binary_info(
521  detail::kernel_info<kernelT>::bin_info,
522  detail::kernel_info<kernelT>::bin_count)
523  .binary_info;
524  const size_t dataSize = binInfo.data_size;
525  detail::binary_address data = binInfo.data;
526  const char* const* const requiredExtensions = binInfo.used_extensions;
527  if (kernelName.empty()) {
528  COMPUTECPP_CL_ERROR_CODE_MSG(
529  0, detail::cpp_error_code::KERNEL_NOT_FOUND_ERROR, nullptr,
530  kernelName.c_str());
531  }
532  if (binInfo.target == nullptr) {
533  COMPUTECPP_CL_ERROR_CODE_MSG(
534  0, detail::cpp_error_code::TARGET_NOT_FOUND_ERROR, nullptr,
535  binInfo.target);
536  }
537 
538  return program(create_program_for_kernel_impl(
539  kernelName.c_str(), data, dataSize, requiredExtensions, c.get_impl(),
540  binInfo.target));
541  }

655  template <typename BIType>
656  static const BIType& select_kernel_binary_info(const BIType* binList,
657  size_t binListSize) {
658  if ((binList == nullptr) || (binListSize == 0)) {
659  COMPUTECPP_CL_ERROR_CODE_MSG(
660  0, detail::cpp_error_code::BINARY_NOT_FOUND_ERROR, nullptr,
661  "Unable to retrieve a binary, is integration header included?");
662  }
663 
664  const auto endOfList = binList + binListSize;
665  const auto targetKernel =
666  std::find_if(binList, endOfList, [](const BIType& kernelBinInfo) {
667  return should_use_binary_info_impl(*kernelBinInfo.binary_info);
668  });
669 
670  if (targetKernel != endOfList) {
671  return *targetKernel;
672  }
673 
674  // The default case is to return the first kernel binary available.
675  return binList[0];
676  }
