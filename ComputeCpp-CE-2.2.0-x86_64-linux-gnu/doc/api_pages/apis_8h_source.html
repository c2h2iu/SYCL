210  using info = detail::kernel_info<kernelName>;
211 
212  if (std::tuple_size<decltype(info::bin_info[0].arguments)>::value > 0) {
213  auto& binInfo = cl::sycl::program::select_kernel_binary_info(
214  info::bin_info, info::bin_count);
215 
216  return detail::functor_arg_descriptor(info{}, binInfo.arguments.data(),
