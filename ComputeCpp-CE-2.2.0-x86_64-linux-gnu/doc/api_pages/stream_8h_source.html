541 template <typename valueT>
542 void append_hexidecimal(const stream& os, const valueT& value) {
543  /* If value is negative, display the minus sign and convert the value to its
544  * absolute value. */
545  auto absValue = detail::abs_value<valueT>::get(os, value);
546 
547  char digits[MAX_HEX_DIGITS] = {0};
548  int currentDigit = 0;
549 
550  valueT recValue = absValue;
551  while (recValue > 0) {
552  digits[currentDigit] = recValue % 16;
553  recValue /= 16;
554  currentDigit++;
555  }
556 
557  os.append_chars(COMPUTECPP_HEX_PREFIX, 2);
558 
559  for (int i = (currentDigit - 1); i >= 0; i--) {
560  char c = (digits[i] <= 9) ? detail::to_char(digits[i])
561  : static_cast<char>('A' + (digits[i] - 10));
562  os.append_chars(&c, 1);
563  }
564 }

572 template <typename valueT>
573 inline int append_integral(const stream& os, const valueT& value,
574  bool useManipulators = true) {
575  if (value == 0) {
576  // A zero is always displayed
577  auto character = detail::to_char(0);
578  constexpr int numDigits = 1;
579  os.append_chars(&character, numDigits);
580  return numDigits;
581  }
582 
583  /* Handle hex manipulator. */
584  if (useManipulators && (os.get_stream_mode() == stream_mode::hex)) {
585  append_hexidecimal(os, value);
586  return 0;
587  }
588 
589  /* If value is negative, display the minus sign and convert the value to its
590  * absolute value. */
591  auto absValue = detail::abs_value<valueT>::get(os, value);
592 
593  /* Create a variable with the original value, that is divided by 10 until it
594  * reaches zero. */
595  valueT devidedValue = absValue;
596 
597  /* Create an array to store the digits retrieved from the value. */
598  valueT digits[MAX_INTEGRAL_DIGITS] = {0};
599 
600  /* Create a counter to record the number of digits retrieved form the value.
601  */
602  int numDigits = 0;
603 
604  /* While the divided value is still greater than zero */
605  while (devidedValue > 0) {
606  /* Calculate a multiplier based on the current digit counter. */
607  valueT multiplier = 10;
608  for (int m = 0; m < numDigits; m++) {
609  multiplier *= 10;
610  }
611 
612  /* Retrieve the digit based on the current digit counter, by dividing by the
613  * multiplier, multiplying by the multiplier, subtracting this from the
614  * original value and then finally dividing by the multiple divided by 10,
615  * reducing the value down to a single digit. */
616  valueT singleDigit =
617  (absValue - ((absValue / multiplier) * multiplier)) / (multiplier / 10);
618 
619  /* assign the digit retrieved. */
620  digits[numDigits] = singleDigit;
621 
622  /* Divided the value again and increment the digit counter. */
623  devidedValue /= 10;
624  numDigits++;
625  }
626 
627  /* Loop over the array of retrieved digits in reverse order, converting each
628  * to a character and appending it. */
629  for (int i = numDigits - 1; i >= 0; i--) {
630  auto integral = static_cast<int>(digits[i]);
631  auto character = detail::to_char(integral);
632  os.append_chars(&character, 1);
633  }
634 
635  /* Return the number fo digits. */
636  return numDigits;
637 }

645 template <typename valueT>
646 inline void append_floating_point(const stream& os, const valueT& value) {
647  /* If value is negative, display the minus sign and convert the value to its
648  * absolute value. */
649  auto absValue = detail::abs_value<valueT>::get(os, value);
650 
651  /* Create variables for storing the scientific base. */
652  int scientificBase = 0;
653 
654  /* Create a variable to storing whether the base is positive or not. */
655  bool possitiveBase = true;
656 
657  /* Handle scientific manipulator. */
658  if (os.get_stream_mode() == stream_mode::scientific) {
659  /* If value is greater than zero. */
660  if (absValue >= 0) {
661  /* Divide the value by 10 until it reaches a single digit on the integral
662  * side. */
663  while (absValue > valueT(10)) {
664  absValue /= valueT(10);
665  scientificBase++;
666  }
667  }
668  /* If value is less than zero. */
669  else {
670  /* Multiply the value by 10 until it reaches a single digit on the
671  * integral side. */
672  while (absValue < valueT(0.1)) {
673  absValue *= valueT(10);
674  scientificBase--;
675  }
676  possitiveBase = false;
677  }
678  }
679 
680  /* Cast to an integer to retrieve the integral part of the floating point. */
681  int integralPart = static_cast<int>(absValue);
682 
683  /* Append the integral part pf the floating point. */
684  int digitsInIntegralPart = append_integral<int>(os, integralPart);
685 
686  /* Append the period sign. */
687  os.append_chars(COMPUTECPP_DECIMAL_POINT, 1);
688 
689  /* Subtract the integral part from the original value to retrieve the
690  * fractional part. */
691  valueT fractionalPart = absValue - static_cast<valueT>(integralPart);
692 
693  /* Create an array to store the digits retrieved from the value. */
694  valueT digits[MAX_FRACTIONAL_DIGITS] = {0.0f};
695 
696  /* Retrieve the precision. */
697  int precision = os.get_precision();
698 
699  /* Calculate digits for fractional side. */
700  int digitsForFractionalSide = precision - digitsInIntegralPart;
701 
702  /* Loop for each digit of precision. */
703  for (int i = 0; i < (digitsForFractionalSide - 1); i++) {
704  /* Create a multiplier based on the current digit. */
705  valueT currentMultiplier = 10.0f;
706  for (int m = 0; m < i; m++) {
707  currentMultiplier *= 10.0f;
708  }
709 
710  /* Multiply the fractional part by the multiplier in order to convert the
711  * current digit in the fractional part to a digit on the integral part. */
712  valueT multipliedFloat = fractionalPart * currentMultiplier;
713 
714  /* Loop for each previously retrieved digit, creating a multiplier for it
715  * and
716  * using that to remove all integral parts beyond the first digit. */
717  for (int r = 0; r < i; r++) {
718  /* Create a multiplier based on the difference in digits between the
719  * current loop and the current digit. */
720  float currentStageMultiplier = 1.0f;
721  for (int sm = 0; sm < (i - r); sm++) {
722  currentStageMultiplier *= 10.0f;
723  }
724 
725  /* Multiply the multiplier by the stored digit to obtain the value to
726  * remove from the integral part. */
727  multipliedFloat -= (digits[r] * currentStageMultiplier);
728  }
729 
730  /* Assign the retrieved digit to the array */
731  digits[i] = static_cast<valueT>((static_cast<int>(multipliedFloat)));
732 
733  /* Convert the digit to an character and append it. */
734  auto integral = static_cast<int>(digits[i]);
735  auto character = detail::to_char(integral);
736  os.append_chars(&character, 1);
737  }
738 
739  /* If scientific manipulator is enabled add the scientific base. */
740  if (os.get_stream_mode() == stream_mode::scientific) {
741  /* Append the scientific notation. */
742  if (possitiveBase) {
743  os.append_chars(COMPUTECPP_POSITIVE_EXP_PREFIX, 2);
744  } else {
745  os.append_chars(COMPUTECPP_NEGATIVE_EXP_PREFIX, 2);
746  }
747 
748  /* Create variable to store the number of digits in the scientific base. */
749  int digitsInScientificBase = 0;
750 
751  /* Divide the scientific base by 10 until it reaches zero to detect the
752  * number of digits. */
753  int tempScientificBase = scientificBase;
754  while (tempScientificBase > 0) {
755  tempScientificBase /= 10;
756  digitsInScientificBase++;
757  }
758 
759  /* Add the correct number of zeros before the scientific base. */
760  for (int sbd = 0; sbd < (3 - digitsInScientificBase); sbd++) {
761  os.append_chars("0", 1);
762  }
763 
764  /* Appen the integral specifying the scientific base. */
765  append_integral(os, scientificBase);
766  }
767 }
