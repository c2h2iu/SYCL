213  template <typename workItemFunctionT>
214  void parallel_for_work_item(range<dimensions> flexibleRange,
215  const workItemFunctionT& func) const {
216  const auto globalRange = this->get_global_range();
217  const auto physicalLocalRange = this->get_local_range();
218  const auto groupID = this->get_id();
219  const auto globalIdBase = physicalLocalRange * groupID;
220 
221  id<dimensions> physicalLocalID;
222 
223  for (size_t itemZ = 0; itemZ < flexibleRange[2]; itemZ++) {
224  physicalLocalID[2] = itemZ % physicalLocalRange[2];
225  for (size_t itemY = 0; itemY < flexibleRange[1]; itemY++) {
226  physicalLocalID[1] = itemY % physicalLocalRange[1];
227  for (size_t itemX = 0; itemX < flexibleRange[0]; itemX++) {
228  physicalLocalID[0] = itemX % physicalLocalRange[0];
229 
230  id<dimensions> localID(detail::index_array(itemX, itemY, itemZ));
231  id<dimensions> globalID = globalIdBase + physicalLocalID;
232 
233  h_item<dimensions> currentItem(detail::h_item_base(
234  detail::item_base(localID, flexibleRange),
235  detail::item_base(physicalLocalID, physicalLocalRange),
236  detail::item_base(globalID, globalRange)));
237 
238  func(currentItem);
239  }
240  }
241  }
