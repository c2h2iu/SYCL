440 template <typename syclInfo, typename syclType, typename clType>
441 syclType get_opencl_info_as_sycl(
442  const typename opencl_info_base<syclInfo>::cl_object& clObject,
443  cl_uint clParam) {
444  /* Create an instance of the SYCL type to be returned. */
445  auto syclValue = syclType();
446 
447  /* Create a variable for the size of the value that will be returned. */
448  size_t size = 0;
449 
450  /* Retrieve the size of the OpenCL info value that will be returned. */
451  get_opencl_info<syclInfo>(clObject, clParam, nullptr, 0, &size);
452 
453  /* If zero size, there is nothing else to be done. */
454  if (size == 0) {
455  return syclValue;
456  }
457 
458  /* Allocate memory for the char* based on the size of the value that will be
459  * returned. */
460  vector_class<char> charVector(size);
461 
462  /* Retrieve the value for the OpenCL info parameter specified by clParam. */
463  get_opencl_info<syclInfo>(clObject, clParam, charVector.data(), size,
464  nullptr);
465 
466  /* Calculate the number of elements that were returned. */
467  size_t numElems = size / sizeof(clType);
468 
469  /* Cast the char* to a pointer to the corresponding OpenCL type. */
470  clType* clPtr = reinterpret_cast<clType*>(charVector.data());
471 
472  /* Convert the pointer to the OpenCL type to the corresponding SYCL type. */
473  syclValue =
474  info_convert<clType*, syclType>::cl_to_sycl(clPtr, numElems, clParam);
475 
476  /* Return the SYCL value. */
477  return syclValue;
478 }

502 template <typename syclInfo, typename syclType, typename clType,
503  cl_uint clParam>
504 syclType get_sycl_info(
505  const typename opencl_info_base<syclInfo>::cl_object& clObject,
506  bool isHost) {
507  /* If the requesting object is in hose mode retrieve the info through the host
508  * info definitions. */
509  if (use_host_info_definitions(isHost)) {
510  /* Retrieve the SYCL info host info specified by clParam. */
511  return sycl_host_info<syclType, clParam>::get();
512  }
513  /* If the requesting object is not in hose mode retrieve the info through the
514  OpenCL info API and convert it to the corresponding SYCL type. */
515  else {
516  return detail::get_opencl_info_as_sycl<syclInfo, syclType, clType>(clObject,
517  clParam);
518  }
519 }

526 inline string_class make_valid_string(const char* clPtr,
527  const size_t numElems) {
528  const auto validString = clPtr && numElems != 0;
529 
530  // return empty string object if the argument is empty
531  if (!validString) {

546 inline std::vector<std::string>
547 info_convert<char*, std::vector<std::string>>::cl_to_sycl(const char* clPtr,
548  size_t numElems,
549  cl_uint clParam) {
550  auto inputString = std::string{clPtr};
551 
552  // Replace semi-colons with spaces as some implementations use semi-colons
553  // and some use spaces.
554  std::replace(std::begin(inputString), std::end(inputString), ';', ' ');
555 
556  // Copy each space delimited substring from the input string into the
557  // result vector.
558  std::istringstream inputStream{inputString};
559 
560  std::vector<std::string> result(
561  std::istream_iterator<std::string>{inputStream},
562  std::istream_iterator<std::string>{});
563 
564  if (!inputStream.eof()) {
565  COMPUTECPP_CL_ERROR_CODE_MSG(
566  CL_SUCCESS, detail::cpp_error_code::GET_INFO_ERROR, nullptr,
567  "Error separating extensions into individual strings.");
568  }
569 
570  return result;
