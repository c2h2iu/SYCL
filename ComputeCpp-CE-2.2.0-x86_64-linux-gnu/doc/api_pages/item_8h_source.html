481  device_event async_work_group_copy(local_ptr<dataT> dest,
482  global_ptr<dataT> src, size_t numElements,
483  size_t srcStride = 1) const {
484 #ifdef __SYCL_DEVICE_ONLY__
485  return device_event(
486  detail::async_work_group_copy(dest, src, numElements, 0));
487 #else // __SYCL_DEVICE_ONLY__

491  *(dest + i) = *(src + srcIter);
